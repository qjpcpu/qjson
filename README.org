* qjson
simple and fast json encode/decode lib.

** unmarshal json bytes to JSONTree

#+begin_src
tree,err := Decode([]byte(`{"a":1}`))
// or
tree := JSONTree{}
err := json.Unmarshal([]byte(`{"a":1}`),&tree)
#+end_src

** marshal JSONTree to bytes

#+begin_src
err := json.Marshal(tree)
#+end_src
